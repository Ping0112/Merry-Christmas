function createPhotoFrame() { photoGroup = new THREE.Group(); // Khung viền vàng const frameMesh = new THREE.Mesh(new THREE.PlaneGeometry(16.5, 12.5), new THREE.MeshBasicMaterial({ color: 0xffd700 })); frameMesh.position.z = -0.1; photoGroup.add(frameMesh); // Chỗ hiển thị ảnh (ĐÃ SỬA: color: 0xffffff) const geo = new THREE.PlaneGeometry(16, 12); // Quan trọng: Phải để màu trắng (0xffffff) thì ảnh mới lên màu được! const mat = new THREE.MeshBasicMaterial({ color: 0xffffff, transparent: true, opacity: 0 }); photoMesh = new THREE.Mesh(geo, mat); photoGroup.add(photoMesh); photoGroup.visible = false; scene.add(photoGroup); }
